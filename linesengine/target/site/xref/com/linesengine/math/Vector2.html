3   /**
4    * Vector2 is a class that represents a single point in 2D space.
5    */
6   
7   public class Vector2 
8   {
9       public float x, y;
10      
11      /**
12       * Create a vector at zero.
13       */
14      public Vector2()
15      {
16          this.x = 0f;
17          this.y = 0f;
18      }
19      
20      /**
21       * Create a new 2D vector.
22       * @param x horizontal amount
23       * @param y vertical amount
24       */
25      public Vector2(float x, float y)
26      {
27          this.x = x;
28          this.y = y;
29      }
30      
31      /**
32       * Create a vector from another vector.
33       * @param v vector to be created from
34       */
35      public Vector2(Vector2 v)
36      {
37          v.x = x;
38          v.y = y;
39      }
40      
41      /**
42       * Adds another vector to this vector.
43       * @param other 
44       */
45      public void add(Vector2 other)
46      {
47           this.x += other.x;
48           this.y += other.y;
49      }
50      
51      /**
52       * Multiplies this vector by a float scalar.
53       * @param multiplier 
54       */
55      public void multiply(float multiplier)
56      {
57          this.x *= multiplier; 
58          this.y *= multiplier;
59      }
60      
61      /**
62       * Gives this vector's length as a scalar.
63       * @return 
64       */
65      public float length()
66      {
67          return (float) Math.sqrt((this.x * this.x) + (this.y * this.y));
68      }
69      
70      /**
71       * Creates a dot product scalar.
72       * @param other the other vector to project this vector on
73       * @return 
74       */
75      public float dotProduct(Vector2 other)
76      {
77          return (this.x * other.x) + (this.y * other.y);
78      }
79      
80      /**
81       * Normalizes a vector's length to one.
82       * @return resulting normalized vector
83       */
84      public Vector2 normalize()
85      {
86          float length = this.length();
87          
88          if(length > 0f)
89          {
90              return new Vector2(this.x /= length, this.y /= length);
91          }
92          else
93          {
94              return this;
95          }
96      }
97      
98      /**
99       * Rotates this vector.
100      * @param angle the amount to rotate by
101      * @param pivot the point to be rotated around
102      */
103     public void rotate(float angle, Vector2 pivot)
104     {
105         Vector2 p = new Vector2(this.x, this.y);
106         //rotation matrix is used here
107         //maybe a matrix math class separately? is it needed?
108         double s = (float) Math.sin(Math.toRadians(angle));
109         double c = (float) Math.cos(Math.toRadians(angle));
110         
111         p.x -= pivot.x;
112         p.y -= pivot.y;
113         
114         double xNew = p.x * c - p.y * s;
115         double yNew = p.x * s + p.y * c;
116         
117         this.x = (float) xNew + pivot.x;
118         this.y = (float) yNew + pivot.y;
119     }
120     
121     @Override
122     public String toString()
123     {
124         float roundedX = Math.round(this.x * 1000.0f) / 1000.0f;
125         float roundedY = Math.round(this.y * 1000.0f) / 1000.0f;
126         return "x: " + roundedX + " | y: " + roundedY;
127     }
128 }
