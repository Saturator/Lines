42      /**

46      public BoxBody(float size)

48          this.tl = new Vector2(-0.5f * size, -0.5f * size);
49          this.tr = new Vector2(0.5f * size, -0.5f * size);
50          this.br = new Vector2(0.5f * size, 0.5f * size);
51          this.bl = new Vector2(-0.5f * size, 0.5f * size);
52          super.position = new Vector2(0f, 0f);
53          super.width = size;
54          super.height = size;
55          rotation = 0f;

58      /**
59       * Scales a box outwards from the center equally.
60       * @param size the size of the resulting box
61       */
62      public void scaleBox(float size)
63      {
64          this.tl = new Vector2((-0.5f * size) + super.position.x, (0.5f * size) + super.position.y);
65          this.tr = new Vector2((0.5f * size) + super.position.x, (0.5f * size) + super.position.y);
66          this.br = new Vector2((0.5f * size) + super.position.x, (-0.5f * size) + super.position.y);
67          this.bl = new Vector2((-0.5f * size) + super.position.x, (-0.5f * size) + super.position.y);
68      }

70      @Override
71      public void move(Vector2 movement)
72      {
73          this.tl = new Vector2(this.tl.x + movement.x, this.tl.y + movement.y);
74          this.tr = new Vector2(this.tr.x + movement.x, this.tr.y + movement.y);
75          this.br = new Vector2(this.br.x + movement.x, this.br.y + movement.y);
76          this.bl = new Vector2(this.bl.x + movement.x, this.bl.y + movement.y);
77          
78          super.position.add(movement);

81      @Override
82      public boolean isColliding(Collidable other)
83      {
84          if(other instanceof CircleBody)
85          {
86              return false;
87          }
88          else if(other instanceof BoxBody && isCollidingWithBox((BoxBody) other))
89          {
90              resolveBoxCollision((BoxBody)other);
91              return true;
92          }
93          return false;
94      }

96      public void resolveBoxCollision(BoxBody other)
97      {            
98          Vector2 vel1 = this.velocity;
99          //this.velocity.multiply(0.8f);
100 
101         vel1.multiply(0.6f);
102         other.velocity.add(vel1);
103         super.velocity.multiply(0.2f);
104 
105         physicsRotation(other);
106             
107         super.hasCollided = true;
108     }

110     public void physicsRotation(BoxBody other)
111     {
112         if(!super.hasRotated)
113         {
114             Vector2 diffMiddle = new Vector2(other.middle().x - this.middle().x, other.middle().y - this.middle().y);
115             Vector2 otherDiff = new Vector2(other.velocity.x, other.velocity.y);
116             otherDiff.add(diffMiddle);
117             float rotAmt = otherDiff.x - diffMiddle.x;
118             if(diffMiddle.y < 0f)
119             {
120                 super.rotation = 0.75f * rotAmt;
121                 other.rotation = 0.75f * rotAmt;
122             }
123             else
124             {
125                 super.rotation = -0.25f * rotAmt;
126                 other.rotation = -0.25f * rotAmt;
127             }
128             super.hasRotated = true;
129             other.hasRotated = true;
130         }
131     }

133     public float[] getMinMax(BoxBody body, Vector2 direction)
134     {
135         //Get the DIRECTION
136         Vector2 axis = direction;
137         Vector2[] points = new Vector2[] {body.tl, body.tr, body.br, body.bl};
138         float thisMin = points[0].dotProduct(axis);
139         float thisMax = points[0].dotProduct(axis);
140         for(Vector2 point : points)
141         {
142             float projection = point.dotProduct(axis);
143             if(projection > thisMax)
144             {
145                 thisMax = projection;
146             }
147             if(projection < thisMin)
148             {
149                 thisMin = projection;
150             }
151         }
152         return new float[] {thisMin, thisMax};
153     }

156     /**
157      * Checks whether this specific box is colliding with another specific box.
158      * @param other
159      * @return is the box colliding or not
160      */
161     public boolean isCollidingWithBox(BoxBody other)
162     {   
163         if(other == this) return false;
164         Vector2 direction = new Vector2
165         (other.middle().x - this.middle().x, other.middle().y - this.middle().y)
166         .normalize();
167         float[] thisMinMax = getMinMax(this, direction);
168         float[] otherMinMax = getMinMax(other, direction);
169         return thisMinMax[1] > otherMinMax[0];
170     }

172     /**
173      * Distance from the center of the box to a specific point.
174      * @param point
175      * @return 
176      */
177     public float distance(Vector2 point)
178     {
179         Vector2 middle = this.middle();
180         return new Vector2(middle.x - point.x, middle.y - point.y).length();
181     }

183     /**
184      * Gets the middle of the box in world space.
185      * @return 
186      */
187     public Vector2 middle()
188     {
189         Line line1 = new Line(tr, tl);
190         Line line2 = new Line(br, bl);
191         Vector2 mid1 = line1.getMidpoint();
192         Vector2 mid2 = line2.getMidpoint();
193         Line line3 = new Line(mid1, mid2);
194         Vector2 middle = line3.getMidpoint();
195         return middle;
196     }

198     /**
199      * Creates collision normals, which are lines perpendicular to the given sides.
200      * @return 
201      */
202     public Line[] getCollisionNormals()
203     {
204         Line[] normals = new Line[4]; //array of lines, = [4][2] list of Vector2's
205         
206         normals[0] = new Line(this.tl, this.tr);
207         normals[1] = new Line(this.tr, this.br);
208         normals[2] = new Line(this.br, this.bl);
209         normals[3] = new Line(this.bl, this.tl);
210         
211         for(Line l : normals)
212         {
213             l.rotate(90);
214         }
215         
216         return normals;
217     }
