3   /**
4    * Line is a class that mathematically represents a line in two dimensional space.
5    * This means that we have two 2D vector points.
6    */
7   
8   public class Line
9   {
10      public Vector2[] points;
11      
12      /**
13       * Creates a new line.
14       * @param p0 start point of the line
15       * @param p1 end point of the line
16       */
17      public Line(Vector2 p0, Vector2 p1)
18      {
19          this.points = new Vector2[2];
20          this.points[0] = p0;
21          this.points[1] = p1;
22      }
23      
24      /**
25       * Gets the point exactly in the middle of the line.
26       * @return the midpoint
27       */
28      public Vector2 getMidpoint()
29      {
30          Vector2 midpoint = new Vector2();
31          for(Vector2 p : this.points)
32          {
33              midpoint.add(p);
34          }
35          return new Vector2(midpoint.x/2, midpoint.y/2);
36      }
37      
38      /**
39       * Rotates the line
40       * @param angle 
41       */
42      public void rotate(float angle)
43      {
44          Vector2 midpoint = this.getMidpoint();
45          Vector2[] normalized = new Vector2[2];
46          normalized[0] = new Vector2(this.points[0].x - midpoint.x, this.points[0].y - midpoint.y);
47          normalized[1] = new Vector2(this.points[1].x - midpoint.x, this.points[1].y - midpoint.y);
48          
49          
50          normalized[0].rotate(angle, midpoint);
51          normalized[1].rotate(angle, midpoint);
52          normalized[0].add(midpoint);
53          normalized[1].add(midpoint);
54          this.points[0] = normalized[0];
55          this.points[1] = normalized[1];
56      }
57      
58      @Override
59      public String toString()
60      {
61          return "(" + this.points[0] + "), (" + this.points[1] + ")";
62      }
63  }
