31      public boolean isCollidingWithAnotherCircle(CircleBody other)
32      {
33          float betweenLength = 
34          new Vector2(other.position.x - this.position.x, other.position.y - this.position.y).length();
35          
36          return (betweenLength - (this.radius + other.radius)) <= 0f;
37      }

42          Vector2 directionalVector = 
43          new Vector2(other.position.x - position.x, other.position.y - position.y);
44          
45          directionalVector.normalize();
