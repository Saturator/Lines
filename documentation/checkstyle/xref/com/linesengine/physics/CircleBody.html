1   package com.linesengine.physics;
2   
3   import com.linesengine.math.Vector2;
4   
5   /**
6    * CircleBody is a child object of PhysicsBody that simulates the movement and collisions for a circle shaped area.
7    */
8   public class CircleBody extends PhysicsBody
9   {
10      protected float radius;
11      
12      /**
13       * Creates a CircleBody with a radius of 25 at position 500x500.
14       */
15      public CircleBody()
16      {
17          this.radius = 25f;
18          super.size = 25f;
19          this.position = new Vector2(500f, 500f);
20      }
21      
22      /**
23       * Creates a CircleBody with a radius of 25 at a given position.
24       * @param position 
25       */
26      public CircleBody(Vector2 position)
27      {
28          this.radius = 25f;
29          super.size = 25f;
30          this.position = position;
31      }
32      
33      /**
34       * Creates a new CircleBody with a given radius at position 500x500.
35       * @param radius
36       */
37      public CircleBody(float radius)
38      {
39          this.radius = radius/2;
40          super.size = radius/2;
41          this.position = new Vector2(500f, 500f);
42      }
43      
44      /**
45       * Creates a new CircleBody with a given radius and position.
46       * @param radius
47       * @param position 
48       */
49      public CircleBody(float radius, Vector2 position)
50      {
51          this.radius = radius/2;
52          super.size = radius/2;
53          this.position = position;
54      }
55      
56      @Override
57      public void move(Vector2 movement)
58      {
59          this.position.add(movement);
60      }  
61      
62      @Override
63      public void rotate(float angle) 
64      {
65      }
66      
67      /**
68       * Checks whether this CircleBody is colliding with the given PhysicsBody.
69       * @param other the other given PhysicsBody.
70       * @return is this CircleBody colliding with another PhysicsBody
71       */
72      @Override
73      public boolean isColliding(PhysicsBody other)
74      {
75          if(this.speed() < 0.025f && other.speed() < 0.025f) return false;
76          
77          if(other instanceof CircleBody && isCollidingWithCircle((CircleBody) other))
78          {
79              resolveCircleCollision((CircleBody) other);
80              return true;
81          }
82          else if(other instanceof BoxBody && isCollidingWithBox((BoxBody) other))
83          {
84              resolveBoxCollision((BoxBody) other);
85              return true;
86          }
87          return false;
88      }
89      
90      /**
91       * Gets the minimum and maximum points of this circle in a given direction.
92       * @param direction the direction of the min and max points
93       * @return the dot product of the two points as a float array
94       */
95      public float[] getMinMax(Vector2 direction)
96      {
97          Vector2 axis = direction.normalize();
98          Vector2 minPt = direction.normalize();
99          minPt.multiply(-radius);
100         minPt.add(this.middle());
101         Vector2 maxPt = direction.normalize();
102         maxPt.multiply(radius);
103         maxPt.add(this.middle());
104         float thisMin = minPt.dotProduct(axis);
105         float thisMax = maxPt.dotProduct(axis);
106         return new float[] {thisMin, thisMax};
107     }
108     
109     /**
110      * Transforms the movement of this CircleBody. 
111      * Also transforms the movement and rotation of the other given BoxBody.
112      * @param other the given BoxBody
113      */
114     public void resolveBoxCollision(BoxBody other)
115     {
116         Vector2 vel1 = this.velocity;
117         vel1.multiply(0.5f);
118         other.velocity.add(vel1);
119         super.velocity.multiply(0.3f);
120         other.physicsRotation(this);
121         super.hasCollided = true;
122     }
123     
124     /**
125      * Checks whether this CircleBody is colliding with a given BoxBody.
126      * @param other the given BoxBody
127      * @return is this CircleBody colliding with the given BoxBody
128      */
129     public boolean isCollidingWithBox(BoxBody other)
130     {
131         Vector2 direction = new Vector2
132         (other.middle().x - this.middle().x, other.middle().y - this.middle().y)
133         .normalize();
134         float[] thisMinMax = getMinMax(direction);
135         float[] otherMinMax = other.getMinMax(other, direction);
136         return thisMinMax[1] > otherMinMax[0];
137     }
138     
139     /**
140      * Gets the middle point of this circle as a vector.
141      * @return
142      */
143     @Override
144     public Vector2 middle()
145     {
146         return new Vector2(position.x+radius, position.y+radius);
147     }
148     
149     /**
150      * Checks whether this CircleBody is colliding with another given CircleBody.
151      * @param other the other CircleBody
152      * @return is this CircleBody colliding with the CircleBody
153      */
154     public boolean isCollidingWithCircle(CircleBody other)
155     {
156         if(other.equals(this)) return false;
157         float distance = Vector2.distance(this.middle(), other.middle());
158         return distance <= other.radius + this.radius;
159     }
160     
161     /**
162      * Resolves the circle on circle collision by adding the resulting velocity changes.
163      * @param other the other circle
164      */
165     public void resolveCircleCollision(CircleBody other)
166     {
167         if(other.equals(this)) return;
168         
169         Vector2 directionalVector = 
170         new Vector2(other.position.x - position.x, other.position.y - position.y);
171         
172         directionalVector = directionalVector.normalize();
173         directionalVector.multiply((this.speed() * -0.2f) + (other.speed() * -0.4f));   
174         this.velocity = directionalVector;
175     }
176 }
