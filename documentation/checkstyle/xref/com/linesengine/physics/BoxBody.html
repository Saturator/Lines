6   /**
7    * BoxBody is a child object of PhysicsBody that simulates collision and movement for a box shaped are
8    */
9   public class BoxBody extends PhysicsBody
10  {
11      public Vector2 tl, tr, br, bl; //top left, top right, bottom right, bottom left
12      
13      /*
14  
15       tl->*-------*<-tr
16           |       |
17           |       |
18           |       | 
19       bl->*-------*<-br
20     
21      */

23      /**
24       * Creates a new BoxBody that is 50*50 by default.
25       */
26      public BoxBody()
27      {
28          this.tl = new Vector2(-50f, -50f);
29          this.tr = new Vector2(50f, -50f);
30          this.br = new Vector2(50f, 50f);
31          this.bl = new Vector2(-50f, 50f);
32          super.position = new Vector2(0f, 0f);
33          super.width = 1f;
34          super.height = 1f;
35          rotation = 0f;
36      }
37      /**
38       * Creates a new box body of given size.
39       * @param size size of the body from the center
40       */
41      public BoxBody(float size)
42      {
43          this.tl = new Vector2(-0.5f * size, 0.5f * size);
44          this.tr = new Vector2(0.5f * size, 0.5f * size);
45          this.br = new Vector2(0.5f * size, -0.5f * size);
46          this.bl = new Vector2(-0.5f * size, -0.5f * size);
47          super.position = new Vector2(0f, 0f);
48          super.width = size;
49          super.height = size;
50          rotation = 0f;
51      }
52      
53      /**
54       * Creates a new box body of given size, at certain position.
55       * @param size size of the body from the center
56       * @param pos starting position of the box
57       */
58      public BoxBody(float size, Vector2 pos)
59      {
60          this.tl = new Vector2((-0.5f * size) + pos.x, (0.5f * size) + pos.y);
61          this.tr = new Vector2((0.5f * size) + pos.x, (0.5f * size) + pos.y);
62          this.br = new Vector2((0.5f * size) + pos.x, (-0.5f * size) + pos.y);
63          this.bl = new Vector2((-0.5f * size) + pos.x, (-0.5f * size) + pos.y);
64          super.position = pos;
65          super.width = size;
66          super.height = size;
67          rotation = 0f;
68      }
69      
70      /**
71       * Scales a box outwards from the center equally.
72       * @param size the size of the resulting box
73       */
74      public void scaleBox(float size)
75      {
76          this.tl = new Vector2((-0.5f * size) + super.position.x, (0.5f * size) + super.position.y);
77          this.tr = new Vector2((0.5f * size) + super.position.x, (0.5f * size) + super.position.y);
78          this.br = new Vector2((0.5f * size) + super.position.x, (-0.5f * size) + super.position.y);
79          this.bl = new Vector2((-0.5f * size) + super.position.x, (-0.5f * size) + super.position.y);
80      }

82      /**
83       * Gets all the four corner points of the box.
84       * @return the points as a Vector2 array
85       */
86      public Vector2[] getPts()
87      {
88          return new Vector2[] {tl, tr, br, bl, this.middle()};
89      }
90      
91      /**
92       * Moves the box by moving all the corner points.
93       * @param movement amount of movement as a vector
94       */
95      @Override
96      public void move(Vector2 movement)
97      {
98          this.tl = new Vector2(this.tl.x + movement.x, this.tl.y + movement.y);
99          this.tr = new Vector2(this.tr.x + movement.x, this.tr.y + movement.y);
100         this.br = new Vector2(this.br.x + movement.x, this.br.y + movement.y);
101         this.bl = new Vector2(this.bl.x + movement.x, this.bl.y + movement.y);
102         
103         super.position.add(movement);
104     }

106     /**
107      * Checks whether this BoxBody is colliding with any given PhysicsBody.
108      * @param other the other PhysicsBody to be checked against
109      * @return is the box colliding or not
110      */
111     @Override
112     public boolean isColliding(PhysicsBody other)
113     {   
114         if(this.speed() < 0.025f && other.speed() < 0.025f) return false;
115         
116         if(other instanceof CircleBody && isCollidingWithCircle((CircleBody) other))
117         {
118             resolveCircleCollision((CircleBody) other);
119             return true;
120         }
121         else if(other instanceof BoxBody && isCollidingWithBox((BoxBody) other))
122         {
123             resolveBoxCollision((BoxBody) other);
124             return true;
125         }
126         return false;
127     }
128     
129     /**
130      * Transforms the velocity and rotation of this object on box collision.
131      * @param other the other BoxBody this one is colliding with
132      */
133     public void resolveBoxCollision(BoxBody other)
134     {            
135         Vector2 vel1 = this.velocity;
136         vel1.multiply(0.6f);
137         
138         other.velocity.add(vel1);
139         super.velocity.multiply(0.2f);
140         super.hasCollided = true;
141         physicsRotation(other);
142     }
143     
144     /**
145      * Transforms the velocity and rotation of this (and the other) object on circle collision.
146      * @param other the CircleBody this BoxBody is colliding with
147      */
148     public void resolveCircleCollision(CircleBody other)
149     {            
150         Vector2 vel1 = this.velocity;
151         vel1.multiply(0.75f);
152         other.velocity.add(vel1);
153         super.velocity.multiply(0.33f);
154         super.hasCollided = true;
155         physicsRotation(other);
156     }
157     
158     /**
159      * A specific check whether this BoxBody is colliding with a CircleBody.
160      * @param other the other given CircleBody
161      * @return is this box colliding with the given CircleBody
162      */
163     public boolean isCollidingWithCircle(CircleBody other)
164     {
165         Vector2 direction = new Vector2
166         (other.middle().x - this.middle().x, other.middle().y - this.middle().y)
167         .normalize();
168         float[] thisMinMax = getMinMax(this, direction);
169         float[] otherMinMax = other.getMinMax(direction);
170         return thisMinMax[1] > otherMinMax[0];
171     }
172     
173     /**
174      * Checks whether this specific box is colliding with another specific box.
175      * @param other the other BoxBody we check if this BoxBody is colliding with
176      * @return is the box colliding or not
177      */
178     public boolean isCollidingWithBox(BoxBody other)
179     {   
180         if(other == this) return false;
181         
182         Vector2 direction = new Vector2
183         (other.middle().x - this.middle().x, other.middle().y - this.middle().y);
184         float[] thisMinMax = getMinMax(this, direction);
185         float[] otherMinMax = getMinMax(other, direction);
186         return thisMinMax[1] > otherMinMax[0];
187     }

189     /**
190      * Calculates the rotational behavior to be applied when two objects collide.
191      * @param other the other body that this one is colliding with
192      */
193     public void physicsRotation(PhysicsBody other)
194     {
195         if(!super.hasRotated)
196         {
197             if(other.speed() < 0.5f) other = this;
198             Vector2 diffMiddle = new Vector2(other.middle().x - this.middle().x, other.middle().y - this.middle().y);
199             float rotAmt;
200             if(Math.abs(other.velocity.x) >= Math.abs(other.velocity.y)) 
201             {
202                 rotAmt = Math.abs(other.velocity.x);
203             }
204             else rotAmt = Math.abs(other.velocity.y);
205             
206             boolean movingRight = false;
207             boolean movingLeft = false;
208             boolean movingUp = false;
209             boolean movingDown = false;
210             if(Math.abs(other.velocity.x) > Math.abs(other.velocity.y))
211             {
212                 if(other.velocity.x < 0f) movingLeft = true;
213                 else movingRight = true;
214             }
215             else if(Math.abs(other.velocity.y) > Math.abs(other.velocity.x))
216             {
217                 if(other.velocity.y < 0f) movingUp = true;
218                 else movingDown = true;
219             }
220             float mult = 0.25f;
221             if(movingUp)
222             {
223                 if(diffMiddle.x < 0f) rotateBoth(mult * rotAmt, other);
224                 else rotateBoth(-mult * rotAmt, other);
225             }
226             else if(movingDown)
227             {
228                 if(diffMiddle.x < 0f) rotateBoth(-mult * rotAmt, other);
229                 else rotateBoth(mult * rotAmt, other);
230             }
231             else if(movingRight)
232             {
233                 if(diffMiddle.y < 0f) rotateBoth(mult * rotAmt, other);
234                 else rotateBoth(-mult * rotAmt, other);
235             }
236             else if(movingLeft)
237             {
238                 if(diffMiddle.y < 0f) rotateBoth(-mult * rotAmt, other);
239                 else rotateBoth(mult * rotAmt, other);
240             }
241             super.hasRotated = true;
242             other.hasRotated = true;
243         }
244     }

246     /**
247      * A shortcut method to rotate two given PhysicsBodys.
248      * @param amt amount of rotation
249      * @param other the other PhysicsBody to be rotated
250      */
251     public void rotateBoth(float amt, PhysicsBody other)
252     {
253         super.rotation = amt;
254         other.rotation = amt;
255     }

257     /**
258      * Gets the minimum and maximum points of the box in any given direction.
259      * @param body the BoxBody of which min and max we get
260      * @param direction the direction of the minimum and maximum
261      * @return minimum and maximum points
262      */
263     public float[] getMinMax(BoxBody body, Vector2 direction)
264     {
265         Vector2 axis = direction.normalize();
266         Vector2[] points = new Vector2[] {body.tl, body.tr, body.br, body.bl};
267         float thisMin = points[0].dotProduct(axis);
268         float thisMax = points[0].dotProduct(axis);
269         for(Vector2 point : points)
270         {
271             float projection = point.dotProduct(axis);
272             if(projection > thisMax)
273             {
274                 thisMax = projection;
275             }
276             if(projection < thisMin)
277             {
278                 thisMin = projection;
279             }
280         }
281         return new float[] {thisMin, thisMax};
282     }

284     /**
285      * Distance from the center of the box to a specific point.
286      * @param point the points of comparison
287      * @return distance to the point of comparison
288      */
289     public float distance(Vector2 point)
290     {
291         Vector2 middle = this.middle();
292         return new Vector2(middle.x - point.x, middle.y - point.y).length();
293     }

295     /**
296      * Gets the middle of the box in world space.
297      * @return 
298      */
299     @Override
300     public Vector2 middle()
301     {
302         Line line1 = new Line(tr, tl);
303         Line line2 = new Line(br, bl);
304         Vector2 mid1 = line1.getMidpoint();
305         Vector2 mid2 = line2.getMidpoint();
306         Line line3 = new Line(mid1, mid2);
307         Vector2 middle = line3.getMidpoint();
308         return middle;
309     }

311     /**
312      * Creates collision normals, which are vectors perpendicular to the given sides.
313      * @return an array that contains all four collision normals
314      */
315     public Vector2[] getCollisionNormals()
316     {
317         Line[] normals = new Line[2]; //array of lines, = [2][2] list of Vector2's 
318         normals[0] = new Line(this.tl, this.tr);
319         normals[1] = new Line(this.tr, this.br);
320         Vector2[] normalVectors = new Vector2[2];
321         Vector2 mid = this.middle();
322         for(int i = 0; i < normals.length; i++)
323         {
324             normalVectors[i] = normals[i].getMidpoint();
325             normalVectors[i].subtract(mid);
326         }
327         
328         return normalVectors;
329     }

332      * Rotates the box.
333      * @param angle the amount of rotation as an angle
334      */
335     @Override
336     public void rotate(float angle)
337     {           
338         //these call the vector2 rotation func, not this one
339         Vector2 mid = this.middle();
340         this.tl.rotate(angle, mid);
341         this.tr.rotate(angle, mid);
342         this.br.rotate(angle, mid);
343         this.bl.rotate(angle, mid);
344     }
