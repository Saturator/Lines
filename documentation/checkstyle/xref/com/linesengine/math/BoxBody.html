3   public class BoxBody extends PhysicsBody
4   {
5       protected Vector2 tl, tr, br, bl; //top left, top right, bottom right, bottom left
6       protected float rotation;
7       
8       /*
9   
10       tl->*----------*<-tr
11           |          |
12           |          |
13           |          | 
14       bl->*----------*<-br
15      
16      */
17      
18      //create a box that is 1x1 by default
19      public BoxBody()
20      {
21          this.tl = new Vector2(-0.5f, 0.5f);
22          this.tr = new Vector2(0.5f, 0.5f);
23          this.br = new Vector2(0.5f, -0.5f);
24          this.bl = new Vector2(-0.5f, -0.5f);
25          this.position = new Vector2(0f, 0f);
26          rotation = 0f;
27      }
28      
29      public BoxBody(Vector2 tl, Vector2 tr, Vector2 br, Vector2 bl)
30      {
31          this.tl = tl;
32          this.tr = tr;
33          this.br = br;
34          this.bl = bl;
35          this.position = this.middle();
36          rotation = 0f;
37      }

39      public BoxBody(float size)
40      {
41          this.tl = new Vector2(-0.5f * size, 0.5f * size);
42          this.tr = new Vector2(0.5f * size, 0.5f * size);
43          this.br = new Vector2(0.5f * size, -0.5f * size);
44          this.bl = new Vector2(-0.5f * size, -0.5f * size);
45          this.position = new Vector2(0f, 0f);
46          rotation = 0f;
47      }
48      
49      public BoxBody(float size, Vector2 position)
50      {
51          this.tl = new Vector2(-0.5f * size, 0.5f * size);
52          this.tr = new Vector2(0.5f * size, 0.5f * size);
53          this.br = new Vector2(0.5f * size, -0.5f * size);
54          this.bl = new Vector2(-0.5f * size, -0.5f * size);
55          this.position = position;
56          rotation = 0f;

59      public void scaleBox(float size)
60      {
61          this.tl.add(new Vector2(-0.5f * size, 0.5f * size));
62          this.tr.add(new Vector2(0.5f * size, 0.5f * size));
63          this.br.add(new Vector2(0.5f * size, -0.5f * size));
64          this.bl.add(new Vector2(-0.5f * size, -0.5f * size));
65      }

69      {
70          this.tl = new Vector2(this.tl.x + movement.x, this.tl.y + movement.y);
71          this.tr = new Vector2(this.tr.x + movement.x, this.tr.y + movement.y);
72          this.br = new Vector2(this.br.x + movement.x, this.br.y + movement.y);

75          this.position.add(movement);

91      public boolean isCollidingWithBox(BoxBody other)
92      {
93          /*
94          
95          *------*
96          | this |
97          |   *--|------*
98          |   |  |      |
99          *------*      |
100             |  other  |
101             *---------*
102         
103         in this case: 
104         (this.tr.x > other.bl.x) && (this.bl.y < other.tr.y)
105         
106         in plain english:
107         this box's upper right corner's x position is larger than
108         the other box's lower left corner's x position
109         and
110         this box's lower left corner's y position is smaller than
111         the other box's upper right corner's y position
112         
113         so the collision returns true
114         */
115         
116         return
117         (this.tr.x > other.bl.x || this.bl.x < other.tr.x)
118         &&
119         (this.tr.y > other.bl.y || this.bl.y < other.tr.y);
120     }

122     public float distance(Vector2 point)
123     {
124         Vector2 middle = this.middle();
125         return new Vector2(middle.x - point.x, middle.y - point.y).length();
126     }
127     
128     public Vector2 middle()
129     {
130         return new Vector2((tr.x + tl.x) / 2, (tr.y + br.y) / 2);
131     }
132     
133     public Line[] getCollisionNormals()
134     {
135         Line[] normals = new Line[4]; //array of lines, = [4][2] list of Vector2's
136         
137         normals[0] = new Line(this.tl, this.tr);
138         normals[1] = new Line(this.tr, this.br);
139         normals[2] = new Line(this.br, this.bl);
140         normals[3] = new Line(this.bl, this.tl);
141         
142         for(Line l : normals)
143         {
144             l.rotate(90);
145         }
146         
147         return normals;
148     }
149     
150     public void rotate(float angle)
151     {           
152         //these call the vector2 rotation func, not this one
153         this.tl.rotate(angle);
154         this.tr.rotate(angle);
155         this.br.rotate(angle);
156         this.bl.rotate(angle);
157         
158         //changes the places of top left dots, etc as the box spins
159         //because as the box spins, the top left and all others also change
160         //we change them as many times as the box has been rotated a full 90 degrees
161         for(float i = this.rotation; i <= angle+this.rotation; i += 90)
162         {
163             Vector2 oldTl = this.tl;
164             this.tl = this.tr;
165             this.tr = this.br;
166             this.br = this.bl;
167             this.bl = oldTl;
168         }
169         
170         //put rotation to 0 every time the box has rotated fully
171         if(this.rotation < 360f)
172         {
173             this.rotation += angle;   
174         }
175         else
176         {
177             this.rotation = angle - 360f;
178         }
179     }
