3   public class Line
4   {
5       public Vector2[] points;
6       
7       public Line(Vector2 p0, Vector2 p1)
8       {
9           this.points = new Vector2[2];
10          this.points[0] = p0;
11          this.points[1] = p1;
12      }
13      
14      public Vector2 getMidpoint()
15      {
16          Vector2 midpoint = new Vector2();
17          for(Vector2 p : this.points)
18          {
19              midpoint.add(p);
20          }
21          return new Vector2(midpoint.x/2, midpoint.y/2);
22      }
23      
24      public void rotate(float angle)
25      {
26          Vector2 midpoint = this.getMidpoint();
27          Vector2[] normalized = new Vector2[2];
28          normalized[0] = new Vector2(this.points[0].x - midpoint.x, this.points[0].y - midpoint.y);
29          normalized[1] = new Vector2(this.points[1].x - midpoint.x, this.points[1].y - midpoint.y);
30          
31          
32          normalized[0].rotate(angle);
33          normalized[1].rotate(angle);
34          normalized[0].add(midpoint);
35          normalized[1].add(midpoint);
36          this.points[0] = normalized[0];
37          this.points[1] = normalized[1];
38      }
39      
40      @Override
41      public String toString()
42      {
43          return "(" + this.points[0] + "), (" + this.points[1] + ")";
44      }
