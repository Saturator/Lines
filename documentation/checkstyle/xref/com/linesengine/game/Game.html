68      public synchronized void start()
69      {
70          this.thread = new Thread(this);
71          this.thread.start();
72          this.running = true;

75      public synchronized void stop()
76      {
77          try
78          {
79             this.thread.join();
80             this.running = false;
81          }
82          catch(Exception e)
83          {
84              e.printStackTrace();
85          }
86      }

88      //TODO: move this to a GameTime class
89      public void run()
90      {
91          long lastTime = System.nanoTime();
92          float amountOfTicks = 60f;
93          float ns = 1000000000 / amountOfTicks;
94          float delta = 0f;
95          long timer = System.currentTimeMillis();
96          int frames = 0;
97          
98          while(running)
99          {
100             long now = System.nanoTime();
101             delta += (now - lastTime) / ns;
102             lastTime = now;
103             
104             while(delta >= 1)
105             {
106                 tick();
107                 delta--;   
108             }
109             if(running)
110             {
111                 render();
112             }
113             
114             frames++;
115             
116             if(System.currentTimeMillis() - timer > 1000)
117             {
118                 timer += 1000;
119                 System.out.println("FPS: " + frames);
120                 System.out.println(this.project.getScene(0).findGameObject("circle1").getPhysicsBody().getVelocity());
121                 frames = 0;
122             }
123         }
124         stop();
125     }

127     private void tick()
128     {
129         //TODO: we only get scene 0 atm, change it after the scene is up and running
130         //tick moves the physics forward when deltaTime is 1, so we have a fixed physics update
131         this.project.getScene(0).tick();
132     }
133     
134     private void render()
135     {
136         BufferStrategy bs = this.getBufferStrategy();
137         if(bs == null)
138         {
139             this.createBufferStrategy(3);
140             return;
141         }
142         
143         Graphics g = bs.getDrawGraphics();
144         g.setColor(Color.black);
145         g.fillRect(0, 0, 1000, 1000);
146         this.project.getScene(0).render(g);
147         g.dispose();
148         bs.show();
149     }
